import javax.validation.constraints.NotNull;

import org.apache.commons.lang3.Validate;
import org.apache.commons.lang3.builder.EqualsBuilder;
import org.apache.commons.lang3.builder.HashCodeBuilder;
import org.apache.commons.lang3.builder.ToStringBuilder;
import org.apache.commons.lang3.builder.ToStringStyle;
import org.apiguardian.api.API;
import org.apiguardian.api.API.Status;

/**
 * The basic implementation of {@code NullableParameters}.
 * 
 * @author <a href="mailto:wamphiry@orne.dev">(w) Iker Hernaez</a>
 * @version 1.0, 2022-12
 * @since 0.1
 */
@API(status=Status.EXPERIMENTAL, since="0.1")
public class NullableParametersImpl
implements NullableParameters, NullableParameters.Builder {

    /** If a {@code null} value is accepted. */
    private boolean nullable = NullableParameters.DEFAULT_NULLABLE;

    /**
     * Empty constructor.
     */
    public NullableParametersImpl() {
        super();
    }

    /**
     * Copy constructor.
     * 
     * @param copy The instance to copy.
     */
    public NullableParametersImpl(
            final @NotNull GenerationParameters copy) {
        super();
        Validate.notNull(copy);
        if (copy instanceof NullableParameters) {
            final NullableParameters tcopy = (NullableParameters) copy;
            this.nullable = tcopy.isNullable();
        }
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public boolean isNullable() {
        return this.nullable;
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public void setNullable(
            final boolean nullable) {
        this.nullable = nullable;
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public @NotNull NullableParametersImpl withNullable(
            final boolean nullable) {
        setNullable(nullable);
        return this;
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public int hashCode() {
        return HashCodeBuilder.reflectionHashCode(this);
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public boolean equals(Object obj) {
        return EqualsBuilder.reflectionEquals(this, obj);
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public String toString() {
        return ToStringBuilder.reflectionToString(this, ToStringStyle.JSON_STYLE);
    }
}
