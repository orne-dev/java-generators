import java.util.UUID;

import javax.validation.constraints.NotNull;

import org.apiguardian.api.API;
import org.apiguardian.api.API.Status;

import dev.orne.test.rnd.AbstractTypedGenerator;
import dev.orne.test.rnd.Priority;

/**
 * Generator of {@code UUID} values.
 * 
 * @author <a href="mailto:wamphiry@orne.dev">(w) Iker Hernaez</a>
 * @version 1.0, 2022-10
 * @since 0.1
 */
@API(status=Status.STABLE, since="0.1")
@Priority(Priority.NATIVE_GENERATORS)
public class UUIDGenerator
extends AbstractTypedGenerator<UUID> {

    /** The default value. */
    public static final UUID DEFAULT_VALUE = UUID.fromString(
            "00000000-0000-0000-00-000000000000");

    /**
     * Creates a new instance.
     */
    public UUIDGenerator() {
        super();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public @NotNull UUID defaultValue() {
        return DEFAULT_VALUE;
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public @NotNull UUID randomValue() {
        return UUID.randomUUID();
    }
}
