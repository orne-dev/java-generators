import javax.validation.constraints.NotNull;
import javax.validation.constraints.Size;

import org.apache.commons.lang3.math.NumberUtils;
import org.apiguardian.api.API;
import org.apiguardian.api.API.Status;

/**
 * Generation parameters extractor that detects {@code Size}
 * constraint annotations and sets {@code SizeParameters.minSize}
 * and {@code SizeParameters.maxSize}.
 * <p>
 * As {@code int} is a primitive type target's {@code minSize} value will
 * be the maximum of it's prior value and sources's value.
 * Same way target's {@code maxSize} value will
 * be the minimum of it's prior value and sources's value.
 * 
 * @author <a href="mailto:wamphiry@orne.dev">(w) Iker Hernaez</a>
 * @version 1.0, 2022-11
 * @since 0.1
 * @see Size
 * @see SizeParameters
 */
@API(status=Status.EXPERIMENTAL, since="0.1")
public class SizeConstraintExtractor
extends AbstractParametersSourceExtractor<SizeParameters, Size> {

    /**
     * Creates a new instance.
     */
    public SizeConstraintExtractor() {
        super();
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public void extractParameters(
            final @NotNull Size from,
            final @NotNull SizeParameters target) {
        target.setMinSize(NumberUtils.max(from.min(), target.getMinSize()));
        target.setMaxSize(NumberUtils.min(from.max(), target.getMaxSize()));
    }
}
