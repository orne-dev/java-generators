25  import javax.validation.constraints.NotNull;
26  
27  import org.apache.commons.lang3.RandomUtils;
28  
29  import dev.orne.test.rnd.AbstractTypedGenerator;
30  import dev.orne.test.rnd.Priority;
31  import dev.orne.test.rnd.generators.EnumGenerator;
32  
33  /**
34   * Custom generator for an enumeration type that prevents generation of an
35   * undesired value.
36   * <p>
37   * Registered through {@code META-INF/services/dev.orne.test.rnd.Generator}
38   * SPI.
39   * <p>
40   * Generator has no {@code Priority} annotation, so gets the default priority
41   * ({@value dev.orne.test.rnd.Priority#DEFAULT}), which is greater than built in
42   * enumeration values generator
43   * ({@value dev.orne.test.rnd.Priority#GENERIC_GENERATORS}) and therefore
44   * takes precedence for generation of the supported types.
45   * 
46   * @author <a href="mailto:wamphiry@orne.dev">(w) Iker Hernaez</a>
47   * @version 1.0, 2022-12
48   * @since 0.1
49   * @see Priority
50   * @see EnumGenerator
51   */
52  public class EnumCustomGenerator
53  extends AbstractTypedGenerator<EnumCustomGenerator.Values> {
54  
55      public static final Values[] GENERATED_VALUES =
56              new Values[] {
57                  Values.DEFAULT,
58                  Values.NON_DEFAULT_A,
59                  Values.NON_DEFAULT_B,
60                  Values.NON_DEFAULT_C,
61              };
62  
63      @Override
64      public @NotNull Values defaultValue() {
65         return Values.DEFAULT;
66      }
67  
68      @Override
69      public @NotNull Values randomValue() {
70          final int index = RandomUtils.nextInt(0, GENERATED_VALUES.length);
71          return GENERATED_VALUES[index];
72      }
73  
74      public static enum Values {
75          DEFAULT,
76          UNDESIRED,
77          NON_DEFAULT_A,
78          NON_DEFAULT_B,
79          NON_DEFAULT_C,
80      }
81  }
