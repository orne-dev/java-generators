54      @Test
55      void testConstructor() {
56          @SuppressWarnings("unchecked")
57          final ParametersSourceExtractor<? super MyParams, ?> extractor1 = spy(ParametersSourceExtractor.class);
58          @SuppressWarnings("unchecked")
59          final ParametersSourceExtractor<MyParams, ?> extractor2 = spy(ParametersSourceExtractor.class);
60          @SuppressWarnings("unchecked")
61          final ParametersSourceExtractor<MyParams, ?> extractor3 = spy(ParametersSourceExtractor.class);
62          assertThrows(NullPointerException.class, () -> {
63              new DefaultParametersExtractor<>(null);
64          });
65          final List<ParametersSourceExtractor<? super MyParams, ?>> extractorsWithNulls = Arrays.asList(
66                  extractor1,
67                  null,
68                  extractor2,
69                  extractor3
70          );
71          assertThrows(IllegalArgumentException.class, () -> {
72              new DefaultParametersExtractor<>(extractorsWithNulls);
73          });
74          final Collection<ParametersSourceExtractor<? super MyParams, ?>> extractors = new ArrayList<>();
75          extractors.add(extractor1);
76          extractors.add(extractor2);
77          final DefaultParametersExtractor<MyParams> result = new DefaultParametersExtractor<>(extractors);
78          assertNotSame(extractors, result.getExtractors());
79          assertEquals(extractors, result.getExtractors());
80          final Collection<ParametersSourceExtractor<? super MyParams, ?>> extractorsBck = new ArrayList<>(extractors);
81          extractors.add(extractor3);
82          assertNotEquals(extractors, result.getExtractors());
83          assertEquals(extractorsBck, result.getExtractors());
84      }

87       * Unit test for {@link DefaultParametersExtractor#extractParameters(Object, Object...)}.
88       */
89      @Test
90      void testExtractParametersVarargs() {
91          final DefaultParametersExtractor<MyParams> generator = spy(new DefaultParametersExtractor<>(
92                  Collections.emptyList()));
93          final MyParams params = mock(MyParams.class);
94          final MySource source1 = mock(MySource.class);
95          final MySource source2 = mock(MySource.class);
96          willDoNothing().given(generator).extractParameters(eq(params), any(Collection.class));
97          assertThrows(NullPointerException.class, () -> {
98              generator.extractParameters(null, (Object[]) null);
99          });
100         assertThrows(NullPointerException.class, () -> {
101             generator.extractParameters(null, source1, source2);
102         });
103         assertThrows(NullPointerException.class, () -> {
104             generator.extractParameters(params, (Object[]) null);
105         });
106         assertThrows(IllegalArgumentException.class, () -> {
107             generator.extractParameters(params, source1, null, source2);
108         });
109         generator.extractParameters(params, source1, source2);
110         then(generator).should().extractParameters(params, Arrays.asList(source1, source2));
111         then(params).shouldHaveNoInteractions();
112         then(source1).shouldHaveNoInteractions();
113         then(source2).shouldHaveNoInteractions();
114     }

124         final MySource source1 = mock(MySource.class);
125         final MySource source2 = mock(MySource.class);

128         assertThrows(NullPointerException.class, () -> {

132             generator.extractParameters(null, sources);
133         });
134         assertThrows(NullPointerException.class, () -> {
135             generator.extractParameters(params, (Collection<?>) null);
136         });
137         final List<Object> sourcesWithNull = Arrays.asList(source1, null, source2);
138         assertThrows(IllegalArgumentException.class, () -> {
139             generator.extractParameters(params, sourcesWithNull);
140         });
141         generator.extractParameters(params, sources);
142         then(generator).should().extract(params, source1);
143         then(generator).should().extract(params, source2);
144         then(params).shouldHaveNoInteractions();
145         then(source1).shouldHaveNoInteractions();
146         then(source2).shouldHaveNoInteractions();

149     /**
150      * Unit test for {@link DefaultParametersExtractor#extract(Object, Object)}.
151      */
152     @Test
153     void testExtract() {
154         @SuppressWarnings("unchecked")
155         final ParametersSourceExtractor<MyParams, MySource> extractor1 = spy(ParametersSourceExtractor.class);
156         willReturn(MySource.class).given(extractor1).getSourceType();
157         @SuppressWarnings("unchecked")
158         final ParametersSourceExtractor<MyParams, MySource> extractor2 = spy(ParametersSourceExtractor.class);
159         willReturn(MySource.class).given(extractor2).getSourceType();
160         @SuppressWarnings("unchecked")
161         final ParametersSourceExtractor<MyParams, MySourceExt> extractor3 = spy(ParametersSourceExtractor.class);
162         willReturn(MySourceExt.class).given(extractor3).getSourceType();
163         final DefaultParametersExtractor<MyParams> generator = spy(new DefaultParametersExtractor<>(
164                 Arrays.asList(extractor1, extractor2, extractor3)));
165         final MyParams params = mock(MyParams.class);
166         final MySourceExt source = mock(MySourceExt.class);
167         generator.extract(params, source);
168         then(extractor1).should(atLeast(0)).getSourceType();
169         then(extractor2).should(atLeast(0)).getSourceType();
170         then(extractor3).should(atLeast(0)).getSourceType();
171         final InOrder order = inOrder(extractor1, extractor2, extractor3);
172         then(extractor3).should(order).extractParameters(source, params);
173         then(extractor2).should(order).extractParameters(source, params);
174         then(extractor1).should(order).extractParameters(source, params);
175         then(extractor1).shouldHaveNoMoreInteractions();
176         then(extractor2).shouldHaveNoMoreInteractions();
177         then(extractor3).shouldHaveNoMoreInteractions();
178     }
